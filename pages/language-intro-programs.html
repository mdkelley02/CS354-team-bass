Build and Run Instructions (For All Programs)

  1. Ensure all dependencies for your platform are installed by following these instructions here.
  2. Copy one of the desired programs below into a file with a .rs extension. For example, linked_list.rs.
  3. From the command line, ensure you are in the directory of the copied program, then run the following commands:

(Windows)


        $ rustc your_program_name.rs
        $ your_program_name.exe

(Mac & Linux)


        $ rustc your_program_name.rs
        $ ./your_program_name

(Author: Senami Hodonu)

      fn seqsum(seq: [u8; 6]) -> u8 {
          //Returns a u8 integer type.
          //u8: 8-bit unsigned integer type
          //The iter() method returns an iterator object of the collection
          //sum() method sums the elements of an iterator.
          //Takes each element, adds them together, and returns the result.
          seq.iter().sum()
      }
      fn main() {
          let seq = [5,6,1,8,3,7];
          println!("Sum: {}", seqsum(seq));
      }

Basic Struct and Enum

(Author: Taylor Brooks)

Running this program will output the following:

Student { name: "Joe", grade: B, age: 20, major: "Mechanical Engineering" }
Student { name: "Joe", grade: A, age: 20, major: "Mechanical Engineering" }


              #[derive(Debug)]
              enum Grade {
                  A,
                  B,
                  C,
                  D,
                  F,
              }
      
              #[derive(Debug)]
              struct Student {
                  name: &'static str,
                  grade: Grade,
                  age: i32,
                  major: String
              }
      
              impl Student {
                  fn changeGrade(&mut self, newGrade: Grade)  {
                      self.grade = newGrade;
                  }
              }
      
              fn main() {
                  let mut student = Student {name: "Joe", grade: Grade::B, age: 20, major: "Mechanical Engineering".to_string()  };
                  println!("{:?}", student);
                  student.changeGrade(Grade::A);
                  println!("{:?}", student);
              }

Basic Lambdas and Higher Order Functions

3628800


              use std::ops::{Mul, RangeInclusive};
      
              fn sum_range(range:RangeInclusive<i32>)->i32 {
                 range.fold(0,|acc,n| acc+n)
              }
      
              fn op_range<F>(range:RangeInclusive<i32>,op:F)->Option<i32> where F:Fn(i32,i32)->i32 {
                 range.reduce(|acc,n| op(acc,n))
              }
      
              fn main() {
                 println!("{}",sum_range(1..=100));
                 println!("{}",match op_range(1..=10,i32::mul) {
                    Some(result) => result,
                    None => panic!("op_range returned an empty value")
                 });
              }

Testing Example

(Author: Xavier Black)

Running this program will output the following:

This test fails thread 'tests::failure' panicked at 'This test fails'
assertion failed: f thread 'tests::false_assertion' panicked at 'assertion failed: f'
Test ignored.


      #[cfg(test)]
        mod tests {
          //Unit testing is built into the language and can be started with #[cfg(test)], a mod tests and the #[test] for each test
          #[test]
           fn it_works() {
            let result = 2 + 2;
            assert_eq!(result, 4);
          }
          //Tests fail if put into panic
          #[test]
          fn failure() {
            panic!("This test fails");
          }
          //Different ways to have a test pass similar to jUnit testing.
          #[test]
          fn asserting() {
            let t = true;
            assert!(t);
          }
          #[test]
          fn false_assertion() {
            let f = false;
            assert!(f);
          }
          //Can disable a test with #[ignore] with a message about why it is being ignored
          #[test]
          #[ignore = "Not done yet"]
          fn too_lazy() {
            panic!("This doesn't matter");
          }
          //Pass if a error should be thrown and can have it match specific error messages
          #[test]
          #[should_panic = "Fire!"]
          fn dumpster_fire() {
            assert_ne!(2,2,"Fire!")
          }
        }
